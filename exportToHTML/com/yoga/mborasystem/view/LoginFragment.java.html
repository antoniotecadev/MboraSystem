96       private void digitarCodigoPin(Integer digito) {
97   
98           digitos.add(String.valueOf(digito));
99   
100          switch (digitos.size()) {
101  
102              case 1:
103                  animarBolinhas(binding.d1);
104                  binding.d1.setTextColor(Color.rgb(111, 55, 0));
105                  break;
106              case 2:
107                  animarBolinhas(binding.d2);
108                  binding.d2.setTextColor(Color.rgb(111, 55, 0));
109                  break;
110              case 3:
111                  animarBolinhas(binding.d3);
112                  binding.d3.setTextColor(Color.rgb(111, 55, 0));
113                  break;
114              case 4:
115                  animarBolinhas(binding.d4);
116                  binding.d4.setTextColor(Color.rgb(111, 55, 0));
117                  break;
118              case 5:
119                  animarBolinhas(binding.d5);
120                  binding.d5.setTextColor(Color.rgb(111, 55, 0));
121                  break;
122              case 6:
123                  animarBolinhas(binding.d6);
124                  binding.d6.setTextColor(Color.rgb(111, 55, 0));
125                  break;
126              default:
127                  break;
128          }
129  
130          if (digitos.size() == 6) {
131              logarComTecladoPersonalizado();
132              limparCodigoPin();
133          }
134      }

155      private void contarTempoDeEspera() {
156          new Thread() {
157              @Override
158              public void run() {
159                  int x = 60;
160                  while (true) {
161                      binding.tvinfoCodigoPin.setText("Tente novamente após " + x-- + " segundos");
162                      if (x == -1) {
163                          return;
164                      }
165  
166                      try {
167                          sleep(1000);
168                      } catch (InterruptedException e) {
169                          e.printStackTrace();
170                          Toast.makeText(getContext(), e.getMessage().toString(), Toast.LENGTH_LONG).show();
171                      }
172  
173                  }
174              }
175  
176          }.start();
177  
178          habilitarTecladoPersonalisado();
179      }

181      private void logarComTecladoPersonalizado() {
182          String codigoPin = "";
183          for (String pin : digitos) {
184              codigoPin += pin;
185          }
186          usuarioViewModel.fazerLogin(Integer.parseInt(codigoPin), getContext());
187      }

197          desabilitarHabilitarButton(false);

245      public void vibrarTelefone(Context context) {
246  
247          if (Build.VERSION.SDK_INT >= 26) {
248              ((Vibrator) context.getSystemService(VIBRATOR_SERVICE)).vibrate(VibrationEffect.createOneShot(500, VibrationEffect.DEFAULT_AMPLITUDE));
249          } else {
250              ((Vibrator) context.getSystemService(VIBRATOR_SERVICE)).vibrate(500);
251          }
